package com.march.example.demo.controller;

import java.util.Scanner;

public class Ali {
    static char[][] ticTacToe = new char[3][3];

    public static void main(String[] args) {

        getInputForEquals();
        fillBoard();
        drawBoard(ticTacToe);
        boolean wonByX = won(ticTacToe, 'X');
        boolean wonByO = won(ticTacToe, 'O');

        if(wonByX){
            System.out.println("X won the game");
        }else if(wonByO){
            System.out.println("O won the game");
        }else{
            System.out.println("It is a draw");
        }
    }

    public static void getInputForEquals() {
        System.out.println("Enter the first list size: ");
        Scanner input = new Scanner(System.in);
        int firstListSize = input.nextInt();
        int[] list1 = new int[firstListSize];//first array
        System.out.println("Enter the first list elements (numbers only): ");

        for (int i = 0; i < firstListSize; i++) {
            list1[i] = Integer.parseInt(input.next());
        }
        System.out.println("Enter the second list size: ");
        int secondListSize = input.nextInt();
        int[] list2 = new int[secondListSize];//first array
        System.out.println("Enter the second list elements (numbers only): ");

        for (int i = 0; i < secondListSize; i++) {
            list2[i] = Integer.parseInt(input.next());
        }

        boolean isEquals = equals(list1, list2);
        if (isEquals) {
            System.out.println("They are equal");
        } else {
            System.out.println("They aren't equal");
        }
    }

    public static boolean equals(int[] list1, int[] list2) {

        if (list1.length > 0 && list2.length > 0) {
            if (list1.length == list2.length) {
                for (int i = 0; i < list1.length; i++) {
                    if (list1[i] != list2[i]) {
                        return false;
                    }
                }
            }
        }
        return true;
    }

    public static void fillBoard() {
        String value = null;
        Scanner input = new Scanner(System.in);
        do {
            System.out.println("Enter 'X' or 'O' with position. Press E to exit!");
            value = input.next();
            if (!value.equalsIgnoreCase("e")) {
                System.out.println("Enter x position (0, 1, 2)");
                int xPosition = input.nextInt();
                System.out.println("Enter y position (0, 1, 2)");
                int yPosition = input.nextInt();
                ticTacToe[xPosition][yPosition] = value.toUpperCase().charAt(0);
            }
        } while (value.equalsIgnoreCase("e"));
    }

    public static void drawBoard(char[][] board) {
        System.out.println("___________");
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[i].length; j++) {
                System.out.println(board[i][j] + "|");
            }
            System.out.println("\n---------");
        }
    }

    public static boolean won(char[][] board, char which) {
        if (board[0][0] == which && board[0][1] == which && board[0][2] == which) {
            return true;
        } else if (board[1][0] == which && board[1][1] == which && board[1][2] == which) {
            return true;
        } else if (board[2][0] == which && board[2][1] == which && board[2][2] == which) {
            return true;
        } else if (board[0][0] == which && board[1][0] == which && board[2][0] == which) {
            return true;
        } else if (board[0][1] == which && board[1][1] == which && board[2][1] == which) {
            return true;
        } else if (board[0][2] == which && board[1][2] == which && board[2][2] == which) {
            return true;
        } else if (board[0][0] == which && board[1][1] == which && board[2][2] == which) {
            return true;
        } else if (board[0][2] == which && board[1][1] == which && board[2][0] == which) {
            return true;
        } else {
            return false;
        }
    }
}
